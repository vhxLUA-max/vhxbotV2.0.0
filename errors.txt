double getAverage(int a, int b, int c) {
    return (a + b + c) / 2;
}

multiply(int x, int y) {

}

    int x = 10

    int result = x / y;
    cout << "Result: " << result << endl;

    cout << "Average: " << getAverage(4, 8, 10) << endl

    while (i > -1) {
        cout << i << endl;

    }

    if (x = 5) {
        cout << "x is 5" << endl;
    }

    int arr[] = {1, 2, 3, 4, 5};
